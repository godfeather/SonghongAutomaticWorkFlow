# 【使用习惯与注意】：
# 	1. 配置文件自动生成，不需要手动创建
# 	2. 文件损坏后直接删掉原来的文件即可，在下一次运行时会生成新的配置文件
#	3. 配置文件中关于流程进行时的配置项可以在流程开始之前配置(流程开始时会重读配置)，不需要重启程序
#	4. 修改配置值时不要改动键（=号左边),以防发生读取为空的情况
#	5. 每行只有一个配置项，每行配置项中不能包含任何空白符
#	6. "#"表示注释符，读取配置时，系统会忽略这些被注释的行，注释时只能注释一整行，并且注释符要在行首第一个位置
#	7. 配置文件严格区分大小写，键遵循小驼峰标志约定俗称#	8. 配置项必须由键值对组成#
driver=firefox
# 测试启动的浏览器，需要特定的驱动支持，建议使用火狐浏览器驱动(firefox)对应驱动名称gecko,可选择chrome浏览器驱动
wait=1000
# 每个节点的等待时间，单位为毫秒；当网络状况不好时可以尝试增加等待时长
tryingCount=60
# 系统判定流程是否被成功审核，如果失败会一直尝试大约{tryingCount}秒，如果在这期间系统判定审核完成，则会继续流程；如果超过尝试次数，系统直接等待用户确认
loginHome=http://t1.itsmore.com:62046/Login/Index
# web登录页，强调的是，必须是登录页的URL不能是首页
rejectStep=1
# 驳回步长：正常情况下，流程走过时所有的审核均给予“同意”，而当需要进行驳回测试时，就会在适当的时候给予节点“不同意”的审批结果，使流程回到发起人；
# 默认情况下，驳回测试每一个节点都会进行一次驳回（取决于rejectStep值为1）,这会花费非常长的时间来审核，而如果需要进行驳回测试，仅进行抽检，
# 则可以设置驳回步长，表示间隔（rejectStep-1）驳回一次
databaseServer=localhost:1433
# 数据库IP:Port
credential=sa/123
# 数据库登录凭据，用户名在前，密码在后，使用“/”隔开
databaseBrand=sqlserver
# 数据库品牌，可选项为mysql和sqlserver，默认为sqlserver,mysql仅支持8.0以上版本
loginTryingCount=5
# 登录确认次数，当登录似乎未成功时，系统将开始尝试确认是否登录成功（是否已进入主页）；尝试设定次数后（1秒确认一次），会停止系统等待用户确认；
# 若登入首页过程中由于网络状况不好而导致超出确认次数，可增加尝试次数
debug=false
# debug模式，如果为true会输出报错信息(并不一定是真意义上的错误，相反，有的错误信息在特定的情况下是需要产生的，以此来驱动另外的逻辑条件)，
# 使用系统时不建议开启
globalPwd=0000
# 所有用户的登录密码，不需要作任何更改
